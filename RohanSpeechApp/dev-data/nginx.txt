server {

    listen 443 ssl default_server;

    listen [::]:443 ssl default_server;



    # SSL configuration

    server_name speechrecognition.ddns.net;

    ssl_certificate /etc/letsencrypt/live/speechrecognition.ddns.net/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/speechrecognition.ddns.net/privkey.pem;


    # Forward requests to the backend server

    location / {

        proxy_pass http://localhost:8080;

        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;

    }

}
